SET(LIB_NAME "UI")

SET(HEADER_PATH ${INCLUDE_PATH}/${LIB_NAME})

INCLUDE_DIRECTORIES(${DEPENDENCIES_PATH}/glfw/include)

FILE(GLOB HEADER_LIST *.h)
FILE(GLOB SOURCE_LIST *.cpp)

SOURCE_GROUP("UI" FILES ${HEADER_LIST})
SOURCE_GROUP("UI" FILES ${SOURCE_LIST})

ADD_EXECUTABLE(${LIB_NAME} ${HEADER_LIST} ${SOURCE_LIST})
TARGET_LINK_LIBRARIES(${LIB_NAME} ray)
TARGET_LINK_LIBRARIES(${LIB_NAME} glfw)

IF(BUILD_PLATFORM_ANDROID)
    TARGET_LINK_LIBRARIES(${LIB_NAME} EGL)
    TARGET_LINK_LIBRARIES(${LIB_NAME} GLESv2)
    TARGET_LINK_LIBRARIES(${LIB_NAME} GLESv3)
ENDIF()
