SET(LIB_NAME "FontGen")

SET(HEADER_PATH ${INCLUDE_PATH}/${LIB_NAME})

FILE(GLOB HEADER_LIST *.h)
FILE(GLOB SOURCE_LIST *.cpp)

INCLUDE_DIRECTORIES(${DEPENDENCIES_PATH}/freetype/include)

SOURCE_GROUP("FontGen" FILES ${HEADER_LIST})
SOURCE_GROUP("FontGen" FILES ${SOURCE_LIST})

ADD_EXECUTABLE(${LIB_NAME} ${HEADER_LIST} ${SOURCE_LIST})
TARGET_LINK_LIBRARIES(${LIB_NAME}
    freetype
    librenderer
)
