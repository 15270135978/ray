SET(LIB_NAME "ui")

SET(HEADER_PATH ${INCLUDE_PATH}/${LIB_NAME})

INCLUDE_DIRECTORIES(${DEPENDENCIES_PATH}/glfw/include)
INCLUDE_DIRECTORIES(${DEPENDENCIES_PATH}/MyGUIEngine/include)
INCLUDE_DIRECTORIES(${DEPENDENCIES_PATH}/MyGUIPlatform/include)

FILE(GLOB HEADER_LIST *.h)
FILE(GLOB SOURCE_LIST *.cpp)

SOURCE_GROUP("ui" FILES ${HEADER_LIST})
SOURCE_GROUP("ui" FILES ${SOURCE_LIST})

ADD_EXECUTABLE(${LIB_NAME} ${HEADER_LIST} ${SOURCE_LIST})
ADD_CUSTOM_COMMAND(TARGET ${LIB_NAME} POST_BUILD COMMAND WORKING_DIRECTORY ${LIBRARY_OUTPUT_PATH})
TARGET_LINK_LIBRARIES(${LIB_NAME}
    ray
    glfw
    glew
    libimage
    libgui
    MyGUIEngine
)
