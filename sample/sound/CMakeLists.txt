SET(LIB_NAME "sound")

SET(HEADER_PATH ${INCLUDE_PATH}/${LIB_NAME})
INCLUDE_DIRECTORIES(${DEPENDENCIES_PATH}/glfw/include)
INCLUDE_DIRECTORIES(${DEPENDENCIES_PATH}/openal/include)
INCLUDE_DIRECTORIES(${DEPENDENCIES_PATH}/libogg)
INCLUDE_DIRECTORIES(${DEPENDENCIES_PATH}/libvorbis)

FILE(GLOB HEADER_LIST *.h)
FILE(GLOB SOURCE_LIST *.cpp)

SOURCE_GROUP("sound" FILES ${HEADER_LIST})
SOURCE_GROUP("sound" FILES ${SOURCE_LIST})

ADD_EXECUTABLE(${LIB_NAME} ${HEADER_LIST} ${SOURCE_LIST})
TARGET_LINK_LIBRARIES(${LIB_NAME} ray glfw libsound)
